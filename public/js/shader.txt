precision highp float;

varying vec2 fragCoord;
uniform vec3 iResolution;
uniform float iGlobalTime;
vec2 iMouse;

/////////////////////////////////////////////////////////////////////////
// Code below (except right at the end) is directly from Shadertoy.com
// https://www.shadertoy.com/view/Xds3zN
/////////////////////////////////////////////////////////////////////////


// Created by inigo quilez - iq/2013
// License Creative Commons Attribution-NonCommercial-ShareAlike 3.0 Unported License.

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    vec2 uv = fragCoord.xy / iResolution.xy;
    fragColor = vec4(0.5+sin(iGlobalTime*1.15-uv.x)*0.5,0.5+sin(iGlobalTime*0.7+uv.y)*0.5,0.5+cos(iGlobalTime*3.27+uv.x*uv.y)*0.5,1.0);
}


/////////////////////////////////////////////////////////////////////////
// Extra code added to integrate Shadertoy.com shader into shaderback.js
/////////////////////////////////////////////////////////////////////////


void main () {
    iMouse = vec2(0.0, 0.0);
    
    mainImage(gl_FragColor, fragCoord);
}
